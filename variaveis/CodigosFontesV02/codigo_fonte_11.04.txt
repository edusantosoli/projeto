      function moveNext()
      {
        x=xmlLivros.recordset;
        if (x.absoluteposition < x.recordcount)
        {
          x.movenext();
        }
      }
      
      function movePrevious()
      {
        x=xmlLivros.recordset;
        if (x.absoluteposition > 1)
        {
          x.moveprevious();
        }
      }
